\begin{Interfaz}
  
    \textbf{se explica con}: \tadNombre{Mapa}
  
    \textbf{géneros}: \TipoVariable{mapa}
  
    \Titulo{Operaciones básicas de mapa}
  
    \InterfazFuncion{crear}{\In{hs}{conj(Nat)},\In{vs}{conj(Nat)}}{mapa}%
    {$res \igobs mapa(hs, vs)$}%
      [$O(copy(hs), copy(vs))$]
    [crea un mapa]

    \InterfazFuncion{esRio}{\In{m1}{Mapa},\In{p}{Pos}}{Bool}%
    {$res \igobs esRio(m1, p)$}%
      [$O(1)$]
    [verifica si en determinada pos hay rio] 

    \InterfazFuncion{Suma}{\In{m1}{Mapa},\In{m2}{Mapa}}{Mapa}%
    {$res \igobs m1 + m2$}%
      [$O(crear(m1) + crear(m2)$]
    [une 2 mapas] 
    \\
  \end{Interfaz}
