    \begin{algorithm}[H]{\textbf{crear}(\In{hs}{conj(Nat)}, \In{vs}{conj(Nat)}) $\to$ $res$ : estr}

        \State $estr.horizontales \gets hs$
        \State $estr.verticales \gets vs$
        \Return estr

    \end{algorithm}
